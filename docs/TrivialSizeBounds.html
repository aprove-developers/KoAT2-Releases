val logger : Batteries.Logger.log

val incoming_bound : [< `Lower | `Upper ] ->
       Program.t ->
       ([ `Lower | `Upper ] -> ProgramTypes.Transition.t -> Var.t -> Bound.t) ->
       LocalSizeBound.t -> ProgramTypes.Transition.t -> Bound.t

Returns the maximum of all incoming sizebounds applied to the local sizebound. Corresponds to 'SizeBounds for trivial SCCs': S'(alpha) = max(S_l(alpha)(S(t',v_1),...,S(t',v_n)) for all t' in pre(t))

val compute : kind ->
       Program.t ->
       ([ `Lower | `Upper ] -> ProgramTypes.Transition.t -> Var.t -> Bound.t) ->
       ProgramTypes.Transition.t * Var.t -> Bound.t

Computes a bound for a trivial scc. That is an scc which consists only of one result variable without a loop to itself. Corresponds to 'SizeBounds for trivial SCCs'.
