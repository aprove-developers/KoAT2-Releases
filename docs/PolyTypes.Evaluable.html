type t

val (=~=) : t -> t -> bool
val equal : t -> t -> bool

Stable structural equality, but not actual equality

val compare : t -> t -> int

Stable structural compare, but not an actual compare

val to_string : ?to_file:bool -> t -> string

Returns a string representing the evaluation. Parameter to_file is used to get a representation with less special characters.

val vars : t -> VarSet.t

Returns a set of the variables which occur in the evaluable

val eval : t ->
       valuation -> value

Assigns each variable inside the polynomial the value of the valuation and returns the arithmetically computed result. !! If the valuation does not provide a value for a variable, an exception is raised. !!

val eval_f : t ->
       (Var.t -> value) -> value

Assigns each variable inside the polynomial the value of the evaluated function and returns the arithmetically computed result.

val rename : RenameMap.t -> t -> t

Assigns the variables of the evaluable new names based on the rename map.

val degree : t -> int

Returns the degree of a polynomial.
