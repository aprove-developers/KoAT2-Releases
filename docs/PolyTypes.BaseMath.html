Module type PolyTypes.BaseMath

module type BaseMath = sig .. end
Modules including BaseMath define basic math operations. They can be typeclass-like extended by MakeMath
type t 
val zero : t
val one : t
val neg : t -> t
val add : t -> t -> t
val mul : t -> t -> t
val pow : t -> int -> t
