Module GraphPrint

module GraphPrint: sig .. end
val print_graph : Fpath.t ->
       string -> 'a -> (Stdlib.Pervasives.out_channel -> 'a -> 'b) -> unit
val print_system : label:(TransitionLabel.t -> string) ->
       outdir:Fpath.t -> file:string -> Program.t -> unit

Prints a png file in the given directory with the given filename (the extension .png will be generated) for the transition graph of the program. For this operation graphviz need to be installed and the 'dot' command must be accessible in the PATH.

val print_rvg : [ `Lower | `Upper ] ->
       label:(ProgramTypes.Transition.t * Var.t -> string) ->
       outdir:Fpath.t -> file:string -> Program.t -> unit

Prints a png file in the given directory with the given filename (the extension .png will be generated) for the result variable graph of the program. For this operation graphviz need to be installed and the 'dot' command must be accessible in the PATH.
